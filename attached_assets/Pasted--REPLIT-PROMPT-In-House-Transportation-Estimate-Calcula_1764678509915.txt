// Use Google Maps Distance Matrix API or similar for precise routing
async function calculateDistance(origin, destination) {
  // Returns: { distanceKm, durationHours, route }
  
  // Fallback to stored matrix if API unavailable
  const storedDistance = distanceMatrix[origin]?.[destination];
  if (storedDistance) return storedDistance;
  
  // API call for custom routes
  const result = await mapsAPI.getDistance(origin, destination);
  return {
    distanceKm: result.distance.value / 1000,
    durationHours: result.duration.value / 3600,
    route: result.routes[0].summary
  };
}

### 3.7 Fuel Surcharge Adjustment

```javascript
// Adjust pricing based on current diesel prices
const BASE_DIESEL_PRICE = 1.65; // CAD per litre (baseline)

function calculateFuelSurcharge(currentDieselPrice, basePrice) {
  const priceDiff = currentDieselPrice - BASE_DIESEL_PRICE;
  const surchargePercent = (priceDiff / BASE_DIESEL_PRICE) * 0.35; // 35% of fuel cost impact
  
  if (surchargePercent > 0) {
    return basePrice * surchargePercent;
  }
  return 0; // No discount if fuel is cheaper (absorb savings)
}
```

## SECTION 4: DELIVERY TIME CALCULATIONS

### 4.1 Transit Time Rules

**Driver Regulations (Canada):**
- Maximum 13 hours driving per day
- Required 8 hours off-duty per 24 hours
- Maximum ~800-1000 km per driver per day

**Transit Time Formula:**
```javascript
function calculateTransitTime(distanceKm, serviceLevel) {
  const kmPerDay = 800; // Conservative estimate
  const baseDays = Math.ceil(distanceKm / kmPerDay);
  
  switch(serviceLevel) {
    case 'STANDARD':
      return {
        pickupWindow: '3-5 business days',
        transitDays: baseDays,
        totalDays: `${3 + baseDays} - ${5 + baseDays} business days`
      };
    
    case 'EXPEDITED_2DAY':
      return {
        pickupWindow: 'Within 48 hours',
        transitDays: baseDays,
        totalDays: `${2 + baseDays} - ${3 + baseDays} days`
      };
    
    case 'EXPEDITED_1DAY':
      return {
        pickupWindow: 'Within 24 hours',
        transitDays: Math.max(baseDays - 1, 1), // Faster routing
        totalDays: `${1 + Math.max(baseDays - 1, 1)} - ${2 + baseDays} days`
      };
  }
}
```

## SECTION 15: TECHNICAL REQUIREMENTS

### 15.1 Database Schema

```sql
-- Transport quotes
CREATE TABLE transport_quotes (
  id SERIAL PRIMARY KEY,
  quote_number VARCHAR(20) UNIQUE,
  user_id INTEGER REFERENCES users(id),
  pickup_address TEXT,
  pickup_city VARCHAR(100),
  pickup_province VARCHAR(2),
  pickup_postal VARCHAR(10),
  delivery_address TEXT,
  delivery_city VARCHAR(100),
  delivery_province VARCHAR(2),
  delivery_postal VARCHAR(10),
  distance_km DECIMAL(10,2),
  vehicle_year INTEGER,
  vehicle_make VARCHAR(50),
  vehicle_model VARCHAR(50),
  vehicle_type VARCHAR(50),
  vehicle_vin VARCHAR(17),
  is_running BOOLEAN DEFAULT TRUE,
  service_level VARCHAR(20), -- STANDARD, EXPEDITED_2DAY, EXPEDITED_1DAY
  base_price DECIMAL(10,2),
  surcharges DECIMAL(10,2),
  discount DECIMAL(10,2),
  total_price DECIMAL(10,2),
  status VARCHAR(20) DEFAULT 'QUOTED',
  valid_until TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- Transport orders
CREATE TABLE transport_orders (
  id SERIAL PRIMARY KEY,
  order_number VARCHAR(20) UNIQUE,
  quote_id INTEGER REFERENCES transport_quotes(id),
  pickup_contact_name VARCHAR(100),
  pickup_contact_phone VARCHAR(20),
  pickup_contact_email VARCHAR(100),
  pickup_instructions TEXT,
  pickup_date DATE,
  pickup_time_preference VARCHAR(20),
  delivery_contact_name VARCHAR(100),
  delivery_contact_phone VARCHAR(20),
  delivery_contact_email VARCHAR(100),
  delivery_instructions TEXT,
  estimated_delivery_date DATE,
  actual_pickup_datetime TIMESTAMP,
  actual_delivery_datetime TIMESTAMP,
  driver_id INTEGER REFERENCES drivers(id),
  truck_id INTEGER REFERENCES trucks(id),
  status VARCHAR(20) DEFAULT 'BOOKED',
  tracking_url VARCHAR(255),
  payment_status VARCHAR(20),
  payment_amount DECIMAL(10,2),
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

-- Fleet trucks
CREATE TABLE trucks (
  id SERIAL PRIMARY KEY,
  unit_number VARCHAR(10) UNIQUE,
  make VARCHAR(50),
  model VARCHAR(50),
  year INTEGER,
  capacity INTEGER, -- vehicles
  current_location VARCHAR(100),
  current_lat DECIMAL(10,8),
  current_lng DECIMAL(11,8),
  status VARCHAR(20), -- AVAILABLE, EN_ROUTE, MAINTENANCE
  last_update TIMESTAMP
);

-- Drivers
CREATE TABLE drivers (
  id SERIAL PRIMARY KEY,
  name VARCHAR(100),
  phone VARCHAR(20),
  email VARCHAR(100),
  license_number VARCHAR(50),
  license_expiry DATE,
  assigned_truck_id INTEGER REFERENCES trucks(id),
  status VARCHAR(20), -- ACTIVE, OFF_DUTY, ON_LEAVE
  photo_url VARCHAR(255)
);

-- Tracking events
CREATE TABLE tracking_events (
  id SERIAL PRIMARY KEY,
  order_id INTEGER REFERENCES transport_orders(id),
  event_type VARCHAR(50),
  location VARCHAR(100),
  lat DECIMAL(10,8),
  lng DECIMAL(11,8),
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```
